    }

    public enum AbilityEffectType
    {
        None = 0,
        FlatDamage = 1,
        FlatHeal = 2,
        DamageOverTime = 3,
        Shield = 4,
        StatBuffNTicks = 5,
        StatDebuffNTicks = 6
    }

    public enum AbilityScale
    {
        CasterAtk = 0,
        CasterMaxHp = 1
    }

    public class AbilityDef
    {
        public string abilityId;
        public string displayName;
        public AbilityTargetType targetType;
        public AbilityEffectType effectType;
        public AbilityScale scale;
        public int magnitudeCoeffMilli;
        public int durationTicks;
        public string description;

        public static AbilityDef NoOp(string id)
        {
            return new AbilityDef
            {
                abilityId = id ?? string.Empty,
                displayName = "None",
                targetType = AbilityTargetType.SelfOnly,
                effectType = AbilityEffectType.None,
                scale = AbilityScale.CasterAtk,
                magnitudeCoeffMilli = 0,
                durationTicks = 0,
                description = "(no-op)"
            };
        }
    }

    public static class AbilityRegistry
    {
        const string CsvFileName = "balance_abilities.csv";
        static readonly Dictionary<string, AbilityDef> Cache = new Dictionary<string, AbilityDef>();
        static bool Loaded;

        public static AbilityDef Get(string abilityId)
        {
            EnsureLoaded();
            if (string.IsNullOrEmpty(abilityId)) return AbilityDef.NoOp(abilityId);
            if (Cache.TryGetValue(abilityId, out var def)) return def;
            Debug.LogWarning($"[AbilityRegistry] unknown abilityId='{abilityId}' → using None");
            var stub = AbilityDef.NoOp(abilityId);
            Cache[abilityId] = stub;
            return stub;
        }

        public static void Reload()
        {
            Cache.Clear();
            Loaded = false;
            EnsureLoaded();
        }

        static void EnsureLoaded()
        {
            if (Loaded) return;
            Loaded = true;
            var rows = CsvLoader.LoadFile(CsvFileName);
            for (int i = 0; i < rows.Count; i++)
            {
                var row = rows[i];
                string id = CsvLoader.GetString(row, "abilityId").Trim();
                if (string.IsNullOrEmpty(id)) continue;

                var def = new AbilityDef
                {
                    abilityId = id,
                    displayName = CsvLoader.GetString(row, "displayName", id),
                    targetType = ParseTarget(CsvLoader.GetString(row, "targetType")),
                    effectType = ParseEffect(CsvLoader.GetString(row, "effectType")),
                    scale = ParseScale(CsvLoader.GetString(row, "scale", "CasterAtk")),
                    magnitudeCoeffMilli = System.Math.Max(0, CsvLoader.GetInt(row, "magnitudeCoeffMilli", 0)),
                    durationTicks = System.Math.Max(0, CsvLoader.GetInt(row, "durationTicks", 0)),
                    description = CsvLoader.GetString(row, "description")
                };
                Cache[id] = def;
            }
            Debug.Log($"[AbilityRegistry] loaded {Cache.Count} abilities");
        }

        static AbilityTargetType ParseTarget(string s)
        {
            if (string.IsNullOrEmpty(s)) return AbilityTargetType.SelfOnly;
            switch (s.Trim())
            {
                case "SelfOnly": return AbilityTargetType.SelfOnly;
                case "FocusedEnemy": return AbilityTargetType.FocusedEnemy;
                case "FirstAliveEnemy": return AbilityTargetType.FirstAliveEnemy;
                case "FirstAliveAlly": return AbilityTargetType.FirstAliveAlly;
                case "LowestHpAliveAlly": return AbilityTargetType.LowestHpAliveAlly;
                case "AllEnemies": return AbilityTargetType.AllEnemies;
                case "AllAllies": return AbilityTargetType.AllAllies;
                default:
                    Debug.LogWarning($"[AbilityRegistry] unknown targetType='{s}' → SelfOnly");
                    return AbilityTargetType.SelfOnly;
            }
        }

        static AbilityEffectType ParseEffect(string s)
        {
            if (string.IsNullOrEmpty(s)) return AbilityEffectType.None;
            switch (s.Trim())
            {
                case "None": return AbilityEffectType.None;
                case "FlatDamage": return AbilityEffectType.FlatDamage;
                case "FlatHeal": return AbilityEffectType.FlatHeal;
                case "DamageOverTime": return AbilityEffectType.DamageOverTime;
                case "Shield": return AbilityEffectType.Shield;
                case "StatBuffNTicks": return AbilityEffectType.StatBuffNTicks;
                case "StatDebuffNTicks": return AbilityEffectType.StatDebuffNTicks;
                default:
                    Debug.LogWarning($"[AbilityRegistry] unknown effectType='{s}' → None");
                    return AbilityEffectType.None;
            }
        }

        static AbilityScale ParseScale(string s)
        {
            if (string.IsNullOrEmpty(s)) return AbilityScale.CasterAtk;
            switch (s.Trim())
            {
                case "CasterAtk": return AbilityScale.CasterAtk;
                case "CasterMaxHp": return AbilityScale.CasterMaxHp;
                default: return AbilityScale.CasterAtk;
            }
        }
    }
}
