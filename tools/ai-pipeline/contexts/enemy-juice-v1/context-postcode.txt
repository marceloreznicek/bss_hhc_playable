✅ Implementado:
- 1.5× escala base (270 px) com clamp `Mathf.Min(270f, (maxW - gap*(n-1))/n)` para caber 8 inimigos no Hollow
- Bob (vertical sin) + weave (horizontal sin) por-frame
- Breathe (uniform scale pulse 0.97–1.03)
- Z-tilt sway (±~3°)
- Hit punch (scale punch 1.18→1.0 sobre 0.12s, ease-out)
- Death wilt (rotaciona 90° + fade alpha sobre 0.4s, terminal)
- Tide pulse (brilho ±10% no inimigo Tide-marker, gated em `isAlive`)
- Healthbar (fill bar acima do sprite) com cor graduada por HP% (verde ≥60, amarelo 30–60, vermelho <30) + shake na bar root no dano
- Phases determinísticas via PCG-hash de `(slotIndex, salt)` (não usa `UnityEngine.Random`, não usa `string.GetHashCode`)
- Per-archetype tuning (`ArchetypeProfile.ForArchetype`): Swarm/Bruiser/Ranged/Caster/Anchor/Grunt
- Hierarquia: `root` (layout, owned by walk-in/walk-out + lunge) → `motionPivot` (child, owned by scheduler) → `Sprite` (Image). `hpBarRoot` é sibling de motionPivot (não filho), para shake independente do bob

⏭️ Diferido para v2 (anotado, não bloqueia v1):
- P7 Caster wind-up — precisa hooks dentro de `RunLiveCombat` para detectar pré-cast; deferred
- Sub-canvas otimização para Image dirty notif — mensurar antes de aplicar

using System.Collections.Generic;
using UnityEngine;
using UnityEngine.UI;

namespace DiverDungeon.Prototype
{
    public enum JuiceMotion
    {
        Idle = 0,   // bob/weave/breathe/tilt active
        Tween = 1,  // attack lunge owned by RunLiveCombat — pause idle motion
        Dead = 2    // wilted, terminal until widget destroyed
    }

    [System.Serializable]
    public class EnemyJuiceState
    {
        public RectTransform motionPivot;     // child of widget root, owned by scheduler
        public Image sprite;                  // color owned by scheduler (composed)
        public RectTransform hpBarRoot;       // sibling of motionPivot, scheduler shakes localPosition
        public Image hpFill;                  // fillAmount + color graded by HP%
        public RectTransform spriteRect;
        public float phaseBobV, phaseBobH, phaseBreathe, phaseTilt, phasePulse;  // set once
        public ArchetypeProfile profile;
        public Color baseSpriteColor = Color.white;
        public Color baseFillColor   = new Color(0.3f, 0.85f, 0.3f, 1f);
        public JuiceMotion motion = JuiceMotion.Idle;
        public float hitPunchUntil;
        public float wiltStartTime;
        public bool  isTide;
        public bool  isAlive = true;
        public float barShakeUntil;
        // Cache p/ pular write redundante de cor
        public float lastFillAmount = -1f;
        public Color lastFillColor = new Color(0,0,0,0);
        public Color lastSpriteColor = new Color(0,0,0,0);
    }

    public struct ArchetypeProfile
    {
        public float bobAmpY, bobFreqY, weaveAmpX, weaveFreqX;
        public float breatheAmp, breatheFreq, tiltDeg, tiltFreq;
        public static ArchetypeProfile ForArchetype(string archetype) { /* per-archetype tuning */ }
    }

    public class EnemyJuiceScheduler : MonoBehaviour
    {
        public List<EnemyJuiceState> states = new List<EnemyJuiceState>();
        public int tideIndex = -1;
        const float TWO_PI = 6.2831853f;
        const float HitPunchDuration = 0.12f, HitPunchPeak = 1.18f;
        const float WiltDuration = 0.4f, DeadAlpha = 0.35f;
        const float TidePulseAmp = 0.10f, TidePulseFreq = 0.55f;
        const float BarShakeDuration = 0.18f, BarShakeAmp = 4f, BarShakeFreq = 32f;

        void LateUpdate()
        {
            float t = Time.unscaledTime;
            for (int i = 0; i < states.Count; i++)
            {
                var s = states[i];
                if (s == null || s.motionPivot == null) continue;
                if (s.motion == JuiceMotion.Tween) continue;  // hard pause; lunge owns transform+color

                Vector2 motionOffset = Vector2.zero;
                float scaleMul = 1f, tiltDeg = 0f;
                Color spriteColor = s.baseSpriteColor;
                bool dead = s.motion == JuiceMotion.Dead;

                if (!dead)
                {
                    float bobY = Mathf.Sin(t * s.profile.bobFreqY * TWO_PI + s.phaseBobV) * s.profile.bobAmpY;
                    float weaveX = Mathf.Sin(t * s.profile.weaveFreqX * TWO_PI + s.phaseBobH) * s.profile.weaveAmpX;
                    motionOffset = new Vector2(weaveX, bobY);
                    scaleMul *= 1f + Mathf.Sin(t * s.profile.breatheFreq * TWO_PI + s.phaseBreathe) * s.profile.breatheAmp;
                    tiltDeg = Mathf.Sin(t * s.profile.tiltFreq * TWO_PI + s.phaseTilt) * s.profile.tiltDeg;
                    if (s.isTide && s.isAlive)
                    {
                        float pulse = 1f + Mathf.Sin(t * TidePulseFreq * TWO_PI + s.phasePulse) * TidePulseAmp;
                        spriteColor = new Color(
                            Mathf.Clamp01(spriteColor.r * pulse),
                            Mathf.Clamp01(spriteColor.g * pulse),
                            Mathf.Clamp01(spriteColor.b * pulse),
                            spriteColor.a);
                    }
                }
                if (!dead && t < s.hitPunchUntil)
                {
                    float u = 1f - Mathf.Clamp01((s.hitPunchUntil - t) / HitPunchDuration);
                    scaleMul *= Mathf.Lerp(HitPunchPeak, 1f, u);
                }
                if (dead)
                {
                    float u = Mathf.Clamp01((t - s.wiltStartTime) / WiltDuration);
                    tiltDeg = Mathf.Lerp(0f, 90f, u);
                    float a = Mathf.Lerp(s.baseSpriteColor.a, DeadAlpha, u);
                    spriteColor = new Color(0.5f, 0.15f, 0.15f, a);
                    motionOffset = Vector2.zero;
                    scaleMul = 1f;
                }

                s.motionPivot.localPosition = new Vector3(motionOffset.x, motionOffset.y, 0f);
                s.motionPivot.localScale = new Vector3(scaleMul, scaleMul, 1f);
                s.motionPivot.localRotation = Quaternion.Euler(0f, 0f, tiltDeg);

                if (s.sprite != null && !ColorsEqual(spriteColor, s.lastSpriteColor))
                {
                    s.sprite.color = spriteColor;
                    s.lastSpriteColor = spriteColor;
                }
                if (s.hpBarRoot != null)
                {
                    if (t < s.barShakeUntil)
                    {
                        float decay = Mathf.Clamp01((s.barShakeUntil - t) / BarShakeDuration);
                        float dx = Mathf.Sin(t * BarShakeFreq) * BarShakeAmp * decay;
                        s.hpBarRoot.localPosition = new Vector3(dx, s.hpBarRoot.localPosition.y, 0f);
                    }
                    else if (s.hpBarRoot.localPosition.x != 0f)
                        s.hpBarRoot.localPosition = new Vector3(0f, s.hpBarRoot.localPosition.y, 0f);
                }
            }
        }

        public static float DeterministicPhase(int slot, int salt)
        {
            unchecked {
                uint x = (uint)(slot + 1) * 747796405u + (uint)salt * 2891336453u;
                x = ((x >> 16) ^ x) * 2246822519u;
                return ((x & 0xFFFFFFu) / 16777216f) * TWO_PI;
            }
        }
    }
}

I4. **Wilt é terminal:** `SetEnemyMotionState` rejeita transições de Dead → outro estado. `SetEnemyHpVisual` em hp<=0 dispara o wilt e desativa hpBarRoot.

I5. **Tide pulse gated em isAlive:** wilt seta isAlive=false; pulse para imediatamente.

I6. **hp/maxHp clamp:** `SetEnemyHpVisual` faz `Mathf.Max(1, maxHp)` e `Mathf.Max(0, currentHp)` antes do divide.

I7. **Phase determinístico cross-platform:** PCG mix usa multiplicações 32-bit unchecked. Sem GetHashCode. Verificar se o overflow é consistente entre IL2CPP/Mono/Editor.
