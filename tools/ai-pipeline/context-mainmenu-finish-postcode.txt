### 3. UIController.WireStaticMainMenu (novo método)
```csharp
void WireStaticMainMenu()
{
    if (staticMainMenuCanvas == null)
    {
        var go = GameObject.Find("MainMenuCanvas");
        if (go != null) staticMainMenuCanvas = go;
    }
    if (staticMainMenuCanvas == null) return;

    if (staticDiveButton == null)
    {
        var t = staticMainMenuCanvas.transform.Find("DiveCenter/DiveButton");
        if (t != null) staticDiveButton = t.GetComponent<Button>();
    }
    if (staticPartyButton == null)
    {
        var t = staticMainMenuCanvas.transform.Find("Navbar/NavParty");
        if (t != null) staticPartyButton = t.GetComponent<Button>();
    }

    if (staticMainMenuWired) return;
    if (staticDiveButton != null)
    {
        staticDiveButton.onClick.RemoveAllListeners();
        staticDiveButton.onClick.AddListener(OnEnterDungeonClicked);
    }
    if (staticPartyButton != null)
    {
        staticPartyButton.onClick.RemoveAllListeners();
        staticPartyButton.onClick.AddListener(OnEnterDungeonClicked);
    }
    staticMainMenuWired = true;
}
```

void RefreshMainMenu()
{
    if (mainMenuStatus == null && menuEnterDungeonBtn == null) return;
    var party = meta.GetParty();
    string partyText = "Party: " + (party.Count == 0 ? "<color=#ff6464>(none — pick one!)</color>" : "");
    for (int i = 0; i < party.Count; i++)
    {
        if (i > 0) partyText += ", ";
        partyText += $"{party[i].heroName} L{party[i].level}";
    }
    if (mainMenuStatus != null)
        mainMenuStatus.text =
            partyText + "\n" +
            $"Banked total: {meta.metaGold} gold\n" +
            $"Runs: {meta.totalRuns}   Deepest: F{meta.deepestFloor}";
    if (menuEnterDungeonBtn != null)
        menuEnterDungeonBtn.interactable = party.Count > 0;
}

### 7. Novo arquivo Assets/Prototype/Scripts/UIPulse.cs
```csharp
using UnityEngine;
using UnityEngine.UI;

namespace DiverDungeon.Prototype
{
    [RequireComponent(typeof(RectTransform))]
    public class UIPulse : MonoBehaviour
    {
        [SerializeField] float duration = 2.4f;
        [SerializeField] float minScale = 0.85f;
        [SerializeField] float maxScale = 1.25f;
        [SerializeField] float startAlpha = 0.7f;
        [SerializeField] float endAlpha = 0f;
        [SerializeField, Range(0f, 1f)] float phaseOffset = 0f;

        Graphic graphic;
        RectTransform rt;
        float baseAlpha;

        void Awake()
        {
            rt = (RectTransform)transform;
            graphic = GetComponent<Graphic>();
            if (graphic != null) baseAlpha = graphic.color.a;
        }

        void OnEnable() => ApplyAt(phaseOffset);

        void Update()
        {
            float t = Mathf.Repeat(Time.unscaledTime / duration + phaseOffset, 1f);
            ApplyAt(t);
        }

        void ApplyAt(float t)
        {
            float s = Mathf.Lerp(minScale, maxScale, t);
            rt.localScale = new Vector3(s, s, 1f);
            if (graphic != null)
            {
                var c = graphic.color;
                c.a = Mathf.Lerp(startAlpha, endAlpha, t);
                graphic.color = c;
            }
        }
    }
}
```
