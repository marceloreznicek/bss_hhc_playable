### JS novo
```js
var GRID_ADJ = [
  /* 0 */ [1, 4],
  /* 1 */ [0, 2, 5],
  /* 2 */ [1, 3, 6],
  /* 3 */ [2, 7],
  /* 4 */ [0, 5],
  /* 5 */ [1, 4, 6],
  /* 6 */ [2, 5, 7],
  /* 7 */ [3, 6]
];
var DRAG = { active: false, srcIdx: -1, pid: null, lastCluster: null, rafQueued: false, lastEv: null };
var CELL_ELS = (function() {
  var arr = new Array(GRID_SIZE);
  var nodes = document.querySelectorAll('.grid-cell');
  for (var i = 0; i < nodes.length; i++) {
    var idx = parseInt(nodes[i].getAttribute('data-idx'), 10);
    if (idx >= 0 && idx < GRID_SIZE) arr[idx] = nodes[i];
  }
  return arr;
})();

function cellIdxFromPoint(x, y) {
  for (var i = 0; i < CELL_ELS.length; i++) {
    var el = CELL_ELS[i];
    if (!el) continue;
    var r = el.getBoundingClientRect();
    if (x >= r.left && x <= r.right && y >= r.top && y <= r.bottom) return i;
  }
  return -1;
}

function computeMergeCluster(srcIdx, dropIdx) {
  if (srcIdx == null || dropIdx == null) return null;
  if (srcIdx < 0 || dropIdx < 0) return null;
  if (srcIdx === dropIdx) return null;
  var s = STATE.grid[srcIdx];
  var d = STATE.grid[dropIdx];
  if (!s || !d) return null;
  if (s.tier !== d.tier) return null;
  if (s.tier >= 5) return null;
  var adj = GRID_ADJ[dropIdx];
  for (var i = 0; i < adj.length; i++) {
    var a = adj[i];
    if (a === srcIdx) continue;
    var u = STATE.grid[a];
    if (u && u.tier === s.tier) return [srcIdx, dropIdx, a];
  }
  return null;
}

function clusterEquals(a, b) {
  if (a === b) return true;
  if (!a || !b) return false;
  if (a.length !== b.length) return false;
  for (var i = 0; i < a.length; i++) if (a[i] !== b[i]) return false;
  return true;
}

function paintPullCluster(cluster) {
  for (var i = 0; i < CELL_ELS.length; i++) {
    var el = CELL_ELS[i];
    if (el && el.classList.contains('pulling')) el.classList.remove('pulling');
  }
  if (!cluster) return;
  for (var k = 0; k < cluster.length; k++) {
    var c = CELL_ELS[cluster[k]];
    if (c) c.classList.add('pulling');
  }
}

function highlightMatches(srcIdx, on) {
  var src = STATE.grid[srcIdx];
  for (var i = 0; i < CELL_ELS.length; i++) {
    var el = CELL_ELS[i];
    if (!el) continue;
    if (i === srcIdx) { el.classList.remove('match'); continue; }
    var u = STATE.grid[i];
    if (on && src && u && u.tier === src.tier) el.classList.add('match');
    else el.classList.remove('match');
  }
}

function clearAllPullStates() {
  for (var i = 0; i < CELL_ELS.length; i++) {
    var el = CELL_ELS[i];
    if (!el) continue;
    el.classList.remove('match');
    el.classList.remove('pulling');
    el.classList.remove('dragging');
  }
}

function moveDragGhost(clientX, clientY) {
  var ghost = document.getElementById('drag-ghost');
  ghost.style.left = clientX + 'px';
  ghost.style.top  = clientY + 'px';
}

function tryMergeCells(srcIdx, dstIdx) {
  var cluster = computeMergeCluster(srcIdx, dstIdx);
  if (!cluster) return false;
  // Revalidate at commit (TOCTOU): tier could have changed between preview and drop.
  var tier = STATE.grid[srcIdx].tier;
  for (var i = 0; i < cluster.length; i++) {
    var u = STATE.grid[cluster[i]];
    if (!u || u.tier !== tier) return false;
  }
  for (var j = 0; j < cluster.length; j++) STATE.grid[cluster[j]] = null;
  STATE.grid[dstIdx] = { tier: tier + 1 };
  STATE.mergesCompleted++;
  playSFX('unlock');
  spawnGridFlash(dstIdx);
  return true;
}

function endDrag() {
  if (!DRAG.active) return;
  clearAllPullStates();
  hideDragGhost();
  DRAG.active = false;
  DRAG.srcIdx = -1;
  DRAG.pid = null;
  DRAG.lastCluster = null;
  DRAG.lastEv = null;
}

function schedulePullUpdate() {
  if (DRAG.rafQueued) return;
  DRAG.rafQueued = true;
  requestAnimationFrame(function() {
    DRAG.rafQueued = false;
    if (!DRAG.active || !DRAG.lastEv) return;
    var ev = DRAG.lastEv;
    var hoverIdx = cellIdxFromPoint(ev.clientX, ev.clientY);
    var cluster = (hoverIdx >= 0) ? computeMergeCluster(DRAG.srcIdx, hoverIdx) : null;
    if (!clusterEquals(cluster, DRAG.lastCluster)) {
      paintPullCluster(cluster);
      DRAG.lastCluster = cluster;
    }
  });
}

document.getElementById('merge-grid').addEventListener('pointerdown', function(ev) {
  if (DRAG.active) return;
  if (!STATE.gameActive || STATE.gameEnded) return;
  var idx = cellIdxFromPoint(ev.clientX, ev.clientY);
  if (idx < 0) return;
  if (!STATE.grid[idx]) return;
  if (STATE.grid[idx].tier >= 5) return;
  DRAG.active = true;
  DRAG.srcIdx = idx;
  DRAG.pid = ev.pointerId;
  DRAG.lastCluster = null;
  DRAG.lastEv = ev;
  if (CELL_ELS[idx]) CELL_ELS[idx].classList.add('dragging');
  highlightMatches(idx, true);
  showDragGhost(idx, ev.clientX, ev.clientY);
  schedulePullUpdate();
  try { ev.target.setPointerCapture && ev.target.setPointerCapture(ev.pointerId); } catch (e) {}
  ev.preventDefault();
});
document.addEventListener('pointermove', function(ev) {
  if (!DRAG.active) return;
  if (ev.pointerId !== DRAG.pid) return;
  moveDragGhost(ev.clientX, ev.clientY);
  DRAG.lastEv = ev;
  schedulePullUpdate();
});
document.addEventListener('pointerup', function(ev) {
  if (!DRAG.active) return;
  if (ev.pointerId !== DRAG.pid) return;
  var srcIdx = DRAG.srcIdx;
  var dstIdx = cellIdxFromPoint(ev.clientX, ev.clientY);
  endDrag();
  if (dstIdx >= 0 && tryMergeCells(srcIdx, dstIdx)) {
    renderGrid();
    tutorialOnFirstMerge();
  }
});
document.addEventListener('pointercancel', function(ev) {
  if (!DRAG.active) return;
  if (ev.pointerId !== DRAG.pid) return;
  endDrag();
});
document.addEventListener('lostpointercapture', function(ev) {
  if (!DRAG.active) return;
  if (ev.pointerId !== DRAG.pid) return;
  endDrag();
});
document.addEventListener('visibilitychange', function() {
  if (document.hidden) endDrag();
});
```

## SPEC DO USUÁRIO (resumido)
1. 3 unidades pra merge, não 2.
2. 2 das 3 (no grid) precisam estar lado a lado (compartilhando borda).
3. Pulsing magnetic-pull mostra quais 3 vão ser consumidas em tempo real.
4. Se houver mais de 3, deixar sobrar (4 → consome 3, sobra 1).
5. Ghost arrastado fica EXATAMENTE embaixo do dedo (não offset).

Procurem regressões no motor legado, edge cases do cluster picker, problemas de animação CSS no `pull` em mobile, e leaks de DOM/listener.
